/**
 * Represents a level in the game.
 */
class Level {
    /**
     * Array of enemies in the level.
     * @type {MovableObject[]}
     */
    enemies;

    /**
     * Array of clouds in the level.
     * @type {MovableObject[]}
     */
    clouds;

    /**
     * Array of background objects in the level.
     * @type {BackroundObject[]}
     */
    backgroundObjects;

    /**
     * Array of coins in the level.
     * @type {Coin[]}
     */
    coins;

    /**
     * Array of bottles in the level.
     * @type {Bottle[]}
     */
    bottles;

    /**
     * The x-coordinate where the level ends.
     * @type {number}
     */
    level_end_x = 2250;

    /**
     * Creates an instance of Level.
     * @param {MovableObject[]} enemies - Array of enemies in the level.
     * @param {MovableObject[]} clouds - Array of clouds in the level.
     * @param {BackroundObject[]} backgroundObjects - Array of background objects in the level.
     * @param {Coin[]} coins - Array of coins in the level.
     * @param {Bottle[]} bottles - Array of bottles in the level.
     */
    constructor(enemies, clouds, backgroundObjects, coins, bottles) {
        this.enemies = enemies;
        this.clouds = clouds;
        this.backgroundObjects = backgroundObjects;
        this.coins = coins;
        this.bottles = bottles;
    }
}
